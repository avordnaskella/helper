name: CI/CD

# Когда запускать
on:
  push:
    branches: [ main, master ]
  pull_request:
    branches: [ main, master ]

# Что делать
jobs:
  # ЗАДАЧА 1: СБОРКА И ТЕСТЫ (CI)
  build-and-test:
    name: Build & Test
    runs-on: ubuntu-latest
    
    steps:
      # 1. Скачать код из репозитория
      - name: Checkout code
        uses: actions/checkout@v4
      
      # 2. Установить .NET
      - name: Setup .NET
        uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '8.0.x'
      
      # 3. Восстановить пакеты
      - name: Restore dependencies
        run: dotnet restore
      
      # 4. Собрать проект
      - name: Build
        run: dotnet build --configuration Release --no-restore
      
      # 5. Запустить тесты
      - name: Run tests
        run: dotnet test --configuration Release --no-build --verbosity normal

  # ЗАДАЧА 2: ПУБЛИКАЦИЯ (CD) - опционально
  # Запускается ТОЛЬКО после успешных тестов и ТОЛЬКО на main
  publish:
    name: Publish
    runs-on: ubuntu-latest
    needs: build-and-test  # Ждёт успешного завершения тестов
    if: github.ref == 'refs/heads/main'  # Только для main
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      
      - name: Setup .NET
        uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '8.0.x'
      
      - name: Publish DLL
        run: dotnet publish CalculatorLibrary/CalculatorLibrary.csproj -c Release -o publish
      
      - name: Upload artifacts
        uses: actions/upload-artifact@v4
        with:
          name: CalculatorLibrary-DLL
          path: publish/
